# log in with shell
# -----------------
# ref: https://docs.influxdata.com/influxdb/v1/tools/shell/
# ref: https://pimylifeup.com/raspberry-pi-influxdb/


# get into the server (make sure influxdb server is running with -- sudo influxd)
influx
influx -h [HOST] -p [PORT] -username [USER] -password

# user 
SHOW USERS
CREATE USER influxDBuser WITH PASSWORD 'influxDBpass' WITH ALL PRIVILEGES
SHOW USERS

# database
SHOW DATABASES
CREATE DATABASE dht
GRANT ALL ON "dht" TO "influxDBuser"
SHOW DATABASES

# exit
exit
