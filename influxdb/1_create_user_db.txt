influx

CREATE USER influxDBuser WITH PASSWORD 'influxDBpass' WITH ALL PRIVILEGES

SHOW USERS

CREATE DATABASE dht

GRANT ALL ON "dht" TO "influxDBuser"

SHOW DATABASES

exit
